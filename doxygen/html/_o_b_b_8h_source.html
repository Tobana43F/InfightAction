35 OBBBBOX CalcBBox(
36 D3DXVECTOR3* vertex,
37 DWORD* index,
38 int numVetices,
39 D3DXMATRIX eigenmat) {
40
41 D3DXVECTOR3 rvec, svec, tvec, vertexvec, center;
42 float rmin = 0.0, rmax = 0.0, smin = 0.0, smax = 0.0, tmin = 0.0, tmax = 0.0;
43 float dot;
44 OBBBBOX result;
45
46 //固有ベクトルが格納された行列を各ベクトルへ代入し正規化
47 rvec.x = eigenmat._11; rvec.y = eigenmat._21; rvec.z = eigenmat._31;
48 svec.x = eigenmat._12; svec.y = eigenmat._22; svec.z = eigenmat._32;
49 tvec.x = eigenmat._13; tvec.y = eigenmat._23; tvec.z = eigenmat._33;
50
51 D3DXVec3Normalize(&rvec, &rvec);
52 D3DXVec3Normalize(&svec, &svec);
53 D3DXVec3Normalize(&tvec, &tvec);
54
55 //各ベクトルとすべての頂点の内積から最小値と最大値を求める
56 for (int i = 0; i<numVetices; i++) {
57 vertexvec.x = vertex[index[i]].x; vertexvec.y = vertex[index[i]].y; vertexvec.z = vertex[index[i]].z;
58 dot = D3DXVec3Dot(&rvec, &vertexvec); // Rベクトルに射影
59 if (i == 0 || dot < rmin) rmin = dot; // 最小値を求める
60 if (i == 0 || dot > rmax) rmax = dot; // 最大値を求める
61
62 dot = D3DXVec3Dot(&svec, &vertexvec); // Sベクトルに射影
63 if (i == 0 || dot < smin) smin = dot; // 最小値を求める
64 if (i == 0 || dot > smax) smax = dot; // 最大値を求める
65
66 dot = D3DXVec3Dot(&tvec, &vertexvec); // Tベクトルに射影
67 if (i == 0 || dot < tmin) tmin = dot; // 最小値を求める
68 if (i == 0 || dot > tmax) tmax = dot; // 最大値を求める
69 }
70
71 //BBOX構造体に値を格納
72 // 中心座標を求める
73 center.x = (rmax + rmin) / 2; center.y = (smax + smin) / 2; center.z = (tmax + tmin) / 2; // 原点からの距離
74 result.center.x = rvec.x * center.x + svec.x * center.y + tvec.x * center.z;
75 result.center.y = rvec.y * center.x + svec.y * center.y + tvec.y * center.z;
76 result.center.z = rvec.z * center.x + svec.z * center.y + tvec.z * center.z;
77
78 result.fLengthX = (rmax - rmin); result.fLengthY = (smax - smin); result.fLengthZ = (tmax - tmin); // BBOXの幅、高さ、奥行
79
80 result.vMax.x = rmax; result.vMax.y = smax; result.vMax.z = tmax; // 最大値の組み合わせ
81 result.vMin.x = rmin; result.vMin.y = smin; result.vMin.z = tmin; // 最小値の組み合わせ
82
83 result.vecAxisX = rvec; result.vecAxisY = svec; result.vecAxisZ = tvec; // BBOXの軸
84
85 return result;
86 }
